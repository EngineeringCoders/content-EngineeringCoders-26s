2.2 Core Terraform Concepts

hcl
123456789101112131415161718192021222324252627282930
# Every Terraform file has this structure:

# 1. PROVIDER — which cloud, which account
provider "aws" {
  region = "ap-south-1"  # Mumbai
}

# 2. RESOURCE — something to create
resource "aws_s3_bucket" "data_lake" {
  bucket = "myname-ai-course-data"
}

# 3. DATA SOURCE — read something that already exists
data "aws_caller_identity" "current" {}

# 4. OUTPUT — expose values for use elsewhere
output "bucket_name" {
  value = aws_s3_bucket.data_lake.bucket
}

# 5. VARIABLE — parameterise your config
variable "environment" {
  type    = string
  default = "dev"
}

# 6. LOCAL — computed values
locals {
  bucket_prefix = "${var.environment}-${data.aws_caller_identity.current.account_id}"
}

Part 3 — Bootstrapping Terraform State (One-Time Setup)

Terminal
bash
12345678910111213141516171819202122232425262728293031323334353637
# bootstrap/setup_state.sh
# Run ONCE before using Terraform for the first time

#!/bin/bash
set -e

STUDENT_NAME=$1
REGION="ap-south-1"
STATE_BUCKET="${STUDENT_NAME}-tf-state-$(aws sts get-caller-identity --query Account --output text)"
TABLE_NAME="tf-state-lock"

echo "Creating state bucket: $STATE_BUCKET"
aws s3 mb "s3://$STATE_BUCKET" --region $REGION

# Enable versioning on state bucket (never lose state!)
aws s3api put-bucket-versioning \
  --bucket "$STATE_BUCKET" \
  --versioning-configuration Status=Enabled

# Enable encryption
aws s3api put-bucket-encryption \
  --bucket "$STATE_BUCKET" \
  --server-side-encryption-configuration '{
    "Rules": [{"ApplyServerSideEncryptionByDefault": {"SSEAlgorithm": "AES256"}}]
  }'

echo "Creating DynamoDB lock table: $TABLE_NAME"
aws dynamodb create-table \
  --table-name "$TABLE_NAME" \
  --attribute-definitions AttributeName=LockID,AttributeType=S \
  --key-schema AttributeName=LockID,KeyType=HASH \
  --billing-mode PAY_PER_REQUEST \
  --region $REGION

echo ""
echo "Done! Update backend config in main.tf:"
echo "  bucket = \"$STATE_BUCKET\""

# src/observability.py
import logging
import os
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.http.trace_exporter import OTLPSpanExporter
from opentelemetry.instrumentation.fastapi import FastAPIInstrumentor
from loguru import logger
import sys

def setup_telemetry(service_name: str) -> trace.Tracer:
    """Configure OpenTelemetry tracing → sends to Grafana Tempo."""
    provider = TracerProvider()
    
    otlp_endpoint = os.getenv("OTEL_EXPORTER_OTLP_ENDPOINT")
    if otlp_endpoint:
        exporter = OTLPSpanExporter(
            endpoint=otlp_endpoint,
            headers={"Authorization": f"Basic {os.getenv('GRAFANA_OTEL_TOKEN')}"}
        )
        provider.add_span_processor(BatchSpanProcessor(exporter))
    
    trace.set_tracer_provider(provider)
    return trace.get_tracer(service_name)

def setup_logging():
    """Structured JSON logging — every log line is machine-parseable."""
    logger.remove()
    logger.add(
        sys.stdout,
        format="{time:YYYY-MM-DD HH:mm:ss.SSS} | {level} | {name}:{line} | {message} | {extra}",
        serialize=True,   # JSON format for log aggregation
        level=os.getenv("LOG_LEVEL", "INFO")
    )
    # Also forward to CloudWatch via Lambda's stdout capture
    return logger

# src/models.py
from __future__ import annotations
from pydantic import BaseModel, field_validator, model_validator
from typing import Annotated
from datetime import datetime
from enum import Enum

# Type aliases make intent clear
CustomerId = Annotated[str, "Customer identifier"]
MoneyAmount = Annotated[float, "Non-negative monetary value in INR"]

class PlanType(str, Enum):
    BASIC    = "basic"
    STANDARD = "standard"
    PREMIUM  = "premium"

class CustomerRecord(BaseModel):
    model_config = {"str_strip_whitespace": True, "frozen": True}

    customer_id:    str
    age:            int
    monthly_spend:  float
    signup_date:    datetime
    plan:           PlanType
    churned:        bool | None = None

    @field_validator("age")
    @classmethod
    def validate_age(cls, v: int) -> int:
        if not 18 <= v <= 120:
            raise ValueError(f"Age {v} outside valid range [18, 120]")
        return v

    @field_validator("monthly_spend")
    @classmethod
    def validate_spend(cls, v: float) -> float:
        if v < 0:
            raise ValueError(f"monthly_spend cannot be negative: {v}")
        return round(v, 2)

class PipelineResult(BaseModel):
    model_config = {"frozen": True}

    run_id:          str
    valid_count:     int
    invalid_count:   int
    duration_ms:     float
    output_s3_key:   str | None = None

    @property
    def total(self) -> int:
        return self.valid_count + self.invalid_count

    @property
    def pass_rate(self) -> float:
        return self.valid_count / self.total if self.total > 0 else 0.0
