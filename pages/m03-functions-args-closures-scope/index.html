2️⃣ Argument Types — The Full Picture

Python
123456789101112131415161718
# ── 1. Positional arguments ───────────────────────
def power(base, exponent):
    return base ** exponent

power(2, 10)    # base=2, exponent=10 (by position)
power(10, 2)    # base=10, exponent=2  ← ORDER MATTERS

# ── 2. Keyword arguments ─────────────────────────
power(base=2, exponent=10)        # same as power(2, 10)
power(exponent=10, base=2)        # order doesn't matter with keywords

# ── 3. Default arguments ─────────────────────────
def connect(host, port=5432, timeout=30):
    print(f"Connecting to {host}:{port} (timeout={timeout}s)")

connect("localhost")                  # port=5432, timeout=30
connect("localhost", 3306)            # port=3306, timeout=30
connect("localhost", timeout=60)      # port=5432, timeout=60

⚠️ Classic gotcha — Mutable Default Arguments

Python
1234567891011121314151617
# WRONG — the list is created ONCE at function definition!
def add_item(item, bag=[]):
    bag.append(item)
    return bag

print(add_item("apple"))    # ['apple']
print(add_item("banana"))   # ['apple', 'banana'] ← BUG! shared list!

# CORRECT — use None as default, create inside function
def add_item(item, bag=None):
    if bag is None:
        bag = []
    bag.append(item)
    return bag

print(add_item("apple"))    # ['apple']
print(add_item("banana"))   # ['banana'] ← correct!

3️⃣ *args and **kwargs

Python
123456789101112131415161718192021222324
# *args — collect extra positional args as a tuple
def total(*numbers):
    print(type(numbers))   # <class 'tuple'>
    return sum(numbers)

print(total(1, 2, 3))          # 6
print(total(10, 20, 30, 40))   # 100

# **kwargs — collect extra keyword args as a dict
def build_profile(**info):
    print(type(info))   # <class 'dict'>
    for key, val in info.items():
        print(f"  {key}: {val}")

build_profile(name="Praveen", role="AI Engineer", city="Hyderabad")

# Combined — the full signature pattern
def mega_function(required, *args, keyword_only="default", **kwargs):
    print(f"required: {required}")
    print(f"extra positional: {args}")
    print(f"keyword_only: {keyword_only}")
    print(f"extra keyword: {kwargs}")

mega_function("must", 1, 2, 3, keyword_only="custom", x=10, y=20)

Unpacking with * and **

Python
12345678910111213
# Spread a list into function arguments
nums = [2, 10]
print(power(*nums))     # same as power(2, 10)

# Spread a dict into function arguments
config = {"host": "localhost", "port": 5432}
connect(**config)       # same as connect(host="localhost", port=5432)

# Merge dicts (Python 3.9+)
defaults = {"color": "red", "size": 10}
custom   = {"size": 20, "weight": 5}
merged   = {**defaults, **custom}
print(merged)   # {'color': 'red', 'size': 20, 'weight': 5}

4️⃣ Positional-Only and Keyword-Only Parameters

Python 3.8+ lets you enforce how arguments must be passed:

Python
1234567891011121314151617
# / means: everything BEFORE this must be positional
# * means: everything AFTER this must be keyword

def strict(pos_only, /, normal, *, kw_only):
    print(pos_only, normal, kw_only)

strict(1, 2, kw_only=3)          # OK
strict(1, normal=2, kw_only=3)   # OK
# strict(pos_only=1, ...)         # TypeError! pos_only can't be keyword
# strict(1, 2, 3)                 # TypeError! kw_only must be keyword

# Real example: len() is positional-only — len(obj=x) doesn't work
# This prevents API breakage when you rename parameters
def calculate(x, y, /, *, rounding=2):
    return round(x + y, rounding)

print(calculate(1.111, 2.222, rounding=1))   # 3.3

5️⃣ Scope — The LEGB Rule

Python resolves names in this exact order: Local → Enclosing → Global → Built-in

Python
1234567891011121314151617
x = "global"      # G — global scope

def outer():
    x = "enclosing"   # E — enclosing scope

    def inner():
        x = "local"   # L — local scope
        print(x)      # "local"

    inner()
    print(x)           # "enclosing"

outer()
print(x)               # "global"

# Built-in (B) — Python's pre-defined names
print(len([1, 2, 3]))   # len is a built-in

global and nonlocal keywords

Python
1234567891011121314151617181920212223
counter = 0

def increment():
    global counter       # reach up to global scope
    counter += 1

increment()
increment()
print(counter)   # 2

# nonlocal — reach up to enclosing (not global) scope
def make_counter():
    count = 0
    def increment():
        nonlocal count   # reach to enclosing scope
        count += 1
        return count
    return increment

my_counter = make_counter()
print(my_counter())   # 1
print(my_counter())   # 2
print(my_counter())   # 3

💡 Rule: Avoid global in production code. It creates hidden dependencies. Use classes, closures, or pass data explicitly.

6️⃣ Closures — Functions That Remember

A closure is a function that captures variables from its enclosing scope, even after that scope has finished executing.

Python
12345678910111213141516171819202122232425262728
def make_multiplier(factor):
    """Returns a function that multiplies by 'factor'"""
    def multiply(number):
        return number * factor    # 'factor' is captured from enclosing scope
    return multiply

double = make_multiplier(2)
triple = make_multiplier(3)

print(double(10))   # 20
print(triple(10))   # 30

# The closure remembers 'factor' even after make_multiplier() finished!
print(double.__closure__[0].cell_contents)   # 2

# Real-world use: configurable functions
def make_validator(min_val, max_val):
    def validate(value):
        if min_val <= value <= max_val:
            return True
        raise ValueError(f"{value} must be between {min_val} and {max_val}")
    return validate

age_validator    = make_validator(0, 120)
salary_validator = make_validator(10000, 10_000_000)

age_validator(25)          # True
salary_validator(500)      # ValueError!

⚠️ The Classic Closure Loop Bug

Python
1234567
# WRONG — all lambdas capture the same 'i' variable
functions = [lambda: i for i in range(5)]
print([f() for f in functions])   # [4, 4, 4, 4, 4]  ← Bug!

# CORRECT — capture by default argument (creates a new binding)
functions = [lambda i=i: i for i in range(5)]
print([f() for f in functions])   # [0, 1, 2, 3, 4]  ← Correct!

7️⃣ Lambda Functions

Anonymous functions for simple, one-expression operations.

Python
1234567891011121314151617181920
# Basic lambda
square = lambda x: x ** 2
print(square(5))   # 25

# Used inline with sorted, map, filter
students = [
    {"name": "Alice", "grade": 88},
    {"name": "Bob",   "grade": 92},
    {"name": "Carol", "grade": 79},
]

# Sort by grade descending
ranked = sorted(students, key=lambda s: s["grade"], reverse=True)
for s in ranked:
    print(f"{s['name']}: {s['grade']}")

# map and filter with lambda
numbers = [1, 2, 3, 4, 5, 6, 7, 8]
evens   = list(filter(lambda x: x % 2 == 0, numbers))
doubled = list(map(lambda x: x * 2, numbers))

⚠️ When NOT to use lambda:
If you need to assign it to a variable and reuse it → use def.
If it needs more than one expression → use def.
Lambda is for short, throwaway functions passed to sorted, map, filter.

✅ Summary

Concept Key Takeaway
Default args Never use mutable defaults — use None and create inside
*args Collects extra positional into a tuple
**kwargs Collects extra keyword into a dict
/ and * Enforce positional-only and keyword-only parameters
LEGB Local → Enclosing → Global → Built-in resolution order
global Avoid it — use closures or classes instead
nonlocal Reach into enclosing scope (not global)
Closures Functions that remember their creation environment
Lambda Short anonymous functions — use sparingly
